#include <cstdlib>
#include <iostream>
using namespace std;

/* float = numero com virgula
 * 
 */
int main(int argc, char** argv) {
    float num1, num2, num3, soma, media;

    cout << "Digite o primeiro numero: ";
    cin >> num1;
    cout << "Digite o segundo numero: ";
    cin >> num2;
    cout << "Digite o terceiro numero: ";
    cin >> num3;

    soma = num1 + num2 + num3;
    media = soma/3;

    cout << "O valor da soma e: " << soma;
    cout << "O valor da media e: " << media << endl;
        
    return 0;
}
